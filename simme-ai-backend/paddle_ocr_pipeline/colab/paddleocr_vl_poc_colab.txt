# 3) Configuracao dos caminhos
# =========================================================
from pathlib import Path
import json
import os
import shutil
import subprocess

os.environ["PADDLE_PDX_DISABLE_MODEL_SOURCE_CHECK"] = "True"

ROOT = Path("/content/drive/MyDrive/paddleocr_vl_poc")
INPUT_DIR = ROOT / "input"
OUTPUT_DIR = ROOT / "output"
LOCAL_INPUT_ROOT = Path("/content/paddleocr_vl_input")

OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
LOCAL_INPUT_ROOT.mkdir(parents=True, exist_ok=True)

print("INPUT_DIR =", INPUT_DIR)
print("OUTPUT_DIR =", OUTPUT_DIR)


# =========================================================
# 4) Escolher documentos
# Deixa DOC_NAMES vazio para correr todas as pastas dentro de input/
# Exemplo para limitar:
# DOC_NAMES = ["CITC_021_CORK SUPPLY PORTUGAL"]
# =========================================================
DOC_NAMES = []
MAX_PAGES_PER_DOC = None  # ex.: 2 para teste rapido; None para todas
STOP_ON_FIRST_ERROR = False
RUN_BY_DOCUMENT = True  # carrega o modelo uma vez por documento; fallback automatico se falhar

if not INPUT_DIR.exists():
    raise RuntimeError(f"A pasta input nao existe no Google Drive: {INPUT_DIR}")

available_doc_dirs = sorted(path for path in INPUT_DIR.iterdir() if path.is_dir())

if DOC_NAMES:
    doc_input_dirs = []
    for doc_name in DOC_NAMES:
        doc_dir = INPUT_DIR / doc_name
        if not doc_dir.exists():
            raise RuntimeError(f"A pasta do documento nao existe: {doc_dir}")
        doc_input_dirs.append(doc_dir)
else:
    doc_input_dirs = available_doc_dirs

if not doc_input_dirs:
    raise RuntimeError(f"Nao existem pastas de documentos em {INPUT_DIR}")

print("Documentos a processar:", len(doc_input_dirs))
for doc_dir in doc_input_dirs:
    print("-", doc_dir.name)


# =========================================================
# 5) Processar cada pasta de input/
# =========================================================
def run_doc_parser(input_path: Path, output_dir: Path) -> subprocess.CompletedProcess:
    cmd = [
        "paddleocr",
        "doc_parser",
        "-i",
        str(input_path),
        "--device",
        DEVICE,
        "--save_path",
        str(output_dir),
    ]

    print("A correr:", " ".join(cmd))
    return subprocess.run(
        cmd,
        capture_output=True,
        text=True,
        encoding="utf-8",
        errors="replace",
    )


def process_document(doc_input_dir: Path) -> dict:
    doc_name = doc_input_dir.name
    doc_output_dir = OUTPUT_DIR / doc_name
    local_doc_dir = LOCAL_INPUT_ROOT / doc_name

    doc_output_dir.mkdir(parents=True, exist_ok=True)

    if local_doc_dir.exists():
        shutil.rmtree(local_doc_dir)
    shutil.copytree(doc_input_dir, local_doc_dir)

    page_names = sorted(
        name
        for name in os.listdir(local_doc_dir)
        if name.startswith("page_") and name.endswith(".png")
    )
    if MAX_PAGES_PER_DOC is not None:
        page_names = page_names[:MAX_PAGES_PER_DOC]

    page_paths = [local_doc_dir / name for name in page_names]

    print("\n" + "=" * 80)
    print("Documento:", doc_name)
    print("Paginas encontradas:", len(page_paths))
    for path in page_paths:
        print("-", path.name)

    all_runs = []

    if not page_paths:
        summary_path = doc_output_dir / "run_summary.json"
        summary_path.write_text("[]", encoding="utf-8")
        return {
            "document": doc_name,
            "pages": 0,
            "success": 0,
            "failed": 0,
            "run_summary": str(summary_path),
            "status": "sem_paginas",
        }

    if RUN_BY_DOCUMENT and MAX_PAGES_PER_DOC is None:
        print("\n=== Modo rapido: documento inteiro ===")
        result = run_doc_parser(local_doc_dir, doc_output_dir)
        all_runs.append(
            {
                "page": "__document__",
                "returncode": result.returncode,
                "stdout": result.stdout,
                "stderr": result.stderr,
                "input_dir": str(local_doc_dir),
                "output_dir": str(doc_output_dir),
                "mode": "document",
            }
        )

        print("returncode =", result.returncode)
        if result.stdout:
            print(result.stdout[:4000])
        if result.stderr:
            print(result.stderr[:4000])

        if result.returncode == 0:
            summary_path = doc_output_dir / "run_summary.json"
            summary_path.write_text(json.dumps(all_runs, ensure_ascii=False, indent=2), encoding="utf-8")
            print("Resumo guardado em:", summary_path)
            return {
                "document": doc_name,
                "pages": len(page_paths),
                "success": len(page_paths),
                "failed": 0,
                "run_summary": str(summary_path),
                "status": "ok",
                "mode": "document",
            }

        print("Modo documento falhou; a tentar fallback pagina-a-pagina.")
        all_runs = []

    for page_path in page_paths:
        page_output_dir = doc_output_dir / page_path.stem
        page_output_dir.mkdir(parents=True, exist_ok=True)

        print(f"\n=== {doc_name} / {page_path.name} ===")
        result = run_doc_parser(page_path, page_output_dir)

        all_runs.append(
            {
                "page": page_path.name,
                "returncode": result.returncode,
                "stdout": result.stdout,
                "stderr": result.stderr,
                "output_dir": str(page_output_dir),
                "mode": "page",
            }
        )

        print("returncode =", result.returncode)
        if result.stdout:
            print(result.stdout[:4000])
        if result.stderr:
            print(result.stderr[:4000])

        if result.returncode != 0 and STOP_ON_FIRST_ERROR:
            print("A parar este documento porque STOP_ON_FIRST_ERROR=True.")
            break

    summary_path = doc_output_dir / "run_summary.json"
    summary_path.write_text(json.dumps(all_runs, ensure_ascii=False, indent=2), encoding="utf-8")

    success = sum(1 for item in all_runs if item["returncode"] == 0)
    failed = sum(1 for item in all_runs if item["returncode"] != 0)

    print("Resumo guardado em:", summary_path)
    print("Sucesso:", success, "| Falhas:", failed)

    return {
        "document": doc_name,
        "pages": len(page_paths),
        "success": success,
        "failed": failed,
        "run_summary": str(summary_path),
        "status": "ok" if failed == 0 else "com_falhas",
        "mode": "page",
    }


# =========================================================
# 6) Correr todos os documentos escolhidos
# =========================================================
all_summaries = []
for doc_input_dir in doc_input_dirs:
    all_summaries.append(process_document(doc_input_dir))

global_summary_path = OUTPUT_DIR / "all_run_summaries.json"
global_summary_path.write_text(
    json.dumps(all_summaries, ensure_ascii=False, indent=2),
    encoding="utf-8",
)

print("\nResumo global guardado em:", global_summary_path)
print(json.dumps(all_summaries, ensure_ascii=False, indent=2))
